\section[Dos Lágrimas]{Dos Lágrimas || \emph{Humberto Ak'abal} \hspace*{\fill}  \thepage}
\label{sec:Dos_Lágrimas.tex}
\vspace*{4cm}
\begin{Parallel}{0.48\textwidth}{0.48\textwidth}
\ParallelLText{\noindent
Cuando nací\\
me pusieron dos lágrimas\\
en los ojos\\
para que pudiera ver\\
el tamaño del dolor de mi gente.}
\ParallelRText{\noindent
When I was born\\
they put two tears \\
in my eyes\\
so that I could see\\
the amount of pain in my people.}
\ParallelPar
\end{Parallel}
